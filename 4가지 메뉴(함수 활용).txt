def sum():
    print("\n1. n까지 정의 합")
    n=int(input("n은?"))
    s=0
    i=0
    while n>i:
        i=i+1
        s=s+i
    print("1부터 n까지의 합계는 %d이다\n"%s) 

def b():
    print("\n2. n~m까지 K의 배수의 합")
    sum0=0
    start=int(input("시작 값을 입력하세요=>"))
    end=int(input("끝 값을 입력하세요=>"))
    b=int(input("배수를 입력하세요=>"))
    for i in range(start,end+1):
        if i%b==0:
            sum0+=i
    print("%d에서 %d까지 %d의 배수의 합은 %d입니다.\n"%(start, end, b, sum0))

def gugudan():
    print("\n3. 구구단 출력")
    n=int(input("몇단?"))
    print("-----%d단-----"%n)
    for i in range(1,10):
        print("%2d * %2d = %2d"%(n,i,n*i))
    print("")
